 HTML Structure
 
  
<!DOCTYPE html>
This document uses html
  
<html lang="en">  
language = english

  <head>
  header containing page titles, links to css, and other things.
  The header is not visible on page

    <meta charset="UTF-8">
    encoding UTF-8
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Webpage title</title>
  </head>

  <body>
   rendering part

    <header>
      <nav>
         nav tag group elements together.
      </nav>
    </header>
